 10 /**
 11  * Class MessageBuilder
 12  * Create messages to send to the server in a simple way.
 13  *
 14  * @author Adrian Gaudebert - adrian@gaudebert.fr, Van-Duc Nguyen
 15  * @constructor
 16  */
 17 function MessageBuilder() {
 18 };
 19 
 20 MessageBuilder.prototype = {
 21 
 22     /**
 23      * Create the basic structure of a message, and stringify to JSON.
 24      * @param type Type of message. Can be "login", "query", "data" or "action".
 25      * @param data Object containing data of the message.
 26      * @return JSON message to send.
 27      */
 28     createMessage: function(type, data) {
 29         var msg = {
 30             type: type,
 31             data: data
 32         };
 33 
 34         return JSON.stringify(msg);
 35     },
 36 
 37     /**
 38      * Create a query message, and stringify to JSON.
 39      * @param responseMethod Type of message we want to receive in response.
 40      * @param responseData Object containing data about this query.
 41      * @return JSON message to send.
 42      */
 43     createQuery: function(responseType, responseData) {
 44         var data = {};
 45         data.response_type = responseType;
 46         data.data = responseData;
 47 
 48         return this.createMessage("query", data);
 49     },
 50 
 51     /**
 52      * Create an action message, and stringify to JSON.
 53      * @param name Name of the action.
 54      * @param data Object containing data about this action.
 55      * @return JSON message to send.
 56      */
 57     createAction: function(name, data) {
 58         var actionData = {};
 59         actionData.name = name;
 60         actionData.data = data;
 61 
 62         return this.createMessage("action", actionData);
 63     },
 64 
 65     /**
 66      * Create a data message, and stringify to JSON.
 67      * @param method What to do with data. Can be "new", "update" or "delete".
 68      * @param object Name of the concerned object.
 69      * @param object_data Object containing data about this object.
 70      * @return JSON message to send.
 71      */
 72     createData: function(method, object, object_data) {
 73         var data = {};
 74         data.method = method;
 75         data.object = object;
 76         data.object_data = object_data;
 77 
 78         return this.createMessage("data", data);
 79     },
 80 
 81     /**
 82      * Create a request message asking for the login of the client.
 83      * @return JSON message to send.
 84      */
 85     createAuthenticationQuery: function() {
 86         return this.createQuery("login", {});
 87     },
 88 
 89     /**
 90      * Create a data message confirming the authentication.
 91      * @param username Login used to authenticate.
 92      * @param valid Boolean to say if authentication is done or not.
 93      * @return JSON message to send.
 94      */
 95     createAuthenticationData: function(userId, valid) {
 96         var data = {};
 97         data.id = userId;
 98         data.valid = valid;
 99         return this.createData("new", "Authentication", data);
100     },
101 
102     createJoinAction: function() {
103         return this.createAction();
104     },
105 
106     createNewGameData: function(game) {
107         return this.createData("new", "Game", game);
108     },
109 
110     createUpdateGameData: function(game) {
111         return this.createData("update", "Game", game);
112     },
113 
114     createNewPlayerData: function(player) {
115         return this.createData('new', 'Player', player);
116     },
117 
118     createUpdatePlayerData: function(player) {
119         return this.createData("update", "player", player);
120     },
121 
122     createUpdateUnitData: function(unit) {
123         return this.createData("update", "Unit", unit);
124     },
125 };
126 
127 module.exports = MessageBuilder;
