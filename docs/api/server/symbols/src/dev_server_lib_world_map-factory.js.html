 10 var map_ = require("./map.js");
 11 var cell_ = require("./cell.js");
 12 var fs = require("fs");
 13 var cellFactory_ = require("./cell-factory.js");
 14 
 15 exports.MapFactory = function() {
 16     this.cellFactory = null;
 17 }
 18 
 19 exports.MapFactory.prototype = {
 20 
 21     create: function() {
 22         return this.createFromFile('data/maps/sample-map.json');
 23     },
 24 
 25     /**
 26      * Create a new map from a map file (JSON format).
 27      * @param file. A JSON file that contains a map object.
 28      * @return myMap.  The created map.
 29      */
 30     createFromFile : function(file) {
 31 
 32         var myMap = new map_.Map();
 33         var file_ = fs.readFileSync(file,"utf8");
 34         var mapObject = JSON.parse(file_);
 35 
 36         myMap.width = mapObject.width;
 37         myMap.height = mapObject.height;
 38         myMap.nbMaxPlayers = mapObject.nbMaxPlayers;
 39 
 40         for (var i = 0; i < myMap.width; i++) {
 41             myMap.cells[i] = [];
 42             for (var j = 0; j < myMap.height; j++) {
 43                 var cell = mapObject.cells[i][j];
 44                 myMap.cells[i][j] = this.cellFactory.create(cell.type, cell.x, cell.y);
 45             }
 46         }
 47 
 48         for (var i = 0; i < mapObject.startPoints.length; i++) {
 49             var startPoint = mapObject.startPoints[i];
 50             myMap.startPoints[i] = myMap.getCell(startPoint.x, startPoint.y);
 51         }
 52 
 53         return myMap;
 54     },
 55 
 56 
 57 }
