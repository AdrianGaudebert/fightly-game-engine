 10 var util = require("util");
 11 
 12 /**
 13  * Class Game
 14  * @author Youness HAMRI - youness.hamri@gmail.com / Van Duc NGUYEN - ducnguyen272@yahoo.com
 15  * @constructor
 16  */
 17 function Game() {
 18 
 19     this.id = null;
 20     this.map = null;
 21     this.players = [];
 22     this.state = "waiting";
 23 
 24     this.mapFactory = null;
 25     this.playerFactory = null;
 26     this.unitFactory = null;
 27 
 28     // Configuration
 29     this.nbMaxPlayers = null;
 30     this.nbMaxTurns = null;
 31     this.turnDuration = null;
 32 
 33 
 34     //play in turn
 35     this.currentPlayer = null;
 36     this.interval = null;
 37     this.currentTurn = null;
 38 };
 39 
 40 Game.prototype = {
 41 
 42     toJSON: function() {
 43         return {
 44             "id":               this.id,
 45             "players":          this.players,
 46             "map":              this.map,
 47             "state":            this.state,
 48             "nbMaxTurns":       this.nbMaxTurns,
 49             "turnDuration":     this.turnDuration,
 50             "currentTurn":      this.currentTurn,
 51             "currentPlayer":    (this.currentPlayer == null) ? null : this.currentPlayer.id,
 52         };
 53     },
 54 
 55     /**
 56      * Adds a new player to current game from an user .
 57      * @param user.
 58      * @return player.
 59      */
 60     addPlayer: function(user) {
 61         util.log("Game.addPlayer");
 62 
 63         var player = this.playerFactory.createFromUser(user, this.map.allocStartPoint());
 64         util.log(player.startPoint);
 65         player.addObserver(this);
 66 
 67         this.players.push(player);
 68         this.notify({game: this, object: "Game", modified: "players", instance: this});
 69 
 70         this.checkState();
 71         return player;
 72     },
 73 
 74     /**
 75      * Set to false a play attribute of player.
 76      * @param PlayerId (a player identity)
 77      */
 78     removePlayer: function(playerId) {
 79         for (var i=0 ; i<this.players.length ; i++) {
 80             if(this.players[i].id == playerId) {
 81                 this.players[i].play = false;
 82                 break;
 83             }
 84         }
 85     },
 86 
 87     /**
 88      * Get a game list of players identities.
 89      * @return PlayersIds
 90      */
 91     getPlayersIds: function() {
 92 
 93         var playersIds = [];
 94         for (i=0 ; i<this.players.length ; i++) {
 95                 playersIds[i] = this.players[i].id;
 96         }
 97         return playersIds;
 98     },
 99 
100     /**
101      * Launchs the game when a maximum number of players is reached
102      * @return this
103      */
104     checkState: function() {
105         if (this.nbMaxPlayers == this.players.length) {
106             this.state = "playing";
107             this.startPlaying();
108             this.notify({game: this, object: "Game", modified: "state", instance: this});
109         }
110         return this;
111     },
112 
113     /**
114      * Downloads one map from a data map file
115      * @param file
116      */
117     downloadMapFromFile: function(file) {
118         this.map = mapFactory.createFromFile(file);
119     },
120 
121     //--->play in turn
122 
123     /**
124      * Get a player by attribute "turn"
125      * @param turn Turn of player
126      * @return player
127      */
128     getPlayerByTurn: function(turn) {
129         var player = null;
130         for (var i = 0; i < this.players.length; i++) {
131             if (this.players[i].turn == turn) {
132                 player = this.players[i];
133                 break;
134             }
135         }
136 
137         return player;
138     },
139 
140 
141     /**
142      * Get player who has next turn
143      * @param currentTurn Turn of current player
144      * @return player
145      */
146     getNextPlayer: function(currentTurn) {
147         var nextPlayer = null;
148         do {
149             currentTurn++;
150             if(currentTurn >= this.nbMaxPlayers) {
151                 currentTurn = 0;
152             }
153 
154             nextPlayer = this.getPlayerByTurn(currentTurn);
155         }
156         while(nextPlayer.play == false);
157 
158         return nextPlayer;
159     },
160 
161 
162     /**
163      * Change turn to the next player (according to Timer)
164      */
165     nextTurn: function() {
166         var nextPlayer = this.getNextPlayer(this.currentPlayer.turn);
167         if(nextPlayer.turn <= this.currentPlayer.turn) {
168             this.currentTurn++;
169             if(this.currentTurn > this.nbMaxTurns) {
170                 this.stopPlaying();
171                 return;
172             }
173         }
174         this.currentPlayer = nextPlayer;
175         this.currentPlayer.resetUnits();
176 
177         this.notify({game: this, object: "Game", modified: "currentPlayer", instance: this});
178         this.notify({game: this, object: "Game", modified: "currentTurn", instance: this});
179 
180         util.log("This is turn of player " + this.currentPlayer.turn + ": " + this.currentPlayer.id);
181 
182         //Start timer for next player
183         this.startTimer();
184     },
185 
186 
187     /**
188      * Start Timer for playing in turn
189      * (in miliseconds)
190      */
191     startTimer: function() {
192         this.interval = setInterval(function() {
193             this.changeTurn();
194         }.bind(this), this.turnDuration * 1000);
195     },
196 
197     /**
198      * Change turn to the next player (immediately)
199      */
200     changeTurn: function() {
201         clearInterval(this.interval);
202         this.nextTurn();
203     },
204 
205 
206     /**
207      * Set the turn order for each player.
208      * @return this.
209      */
210     orderPlayers: function() {
211         var i = 0,
212             l = this.players.length;
213 
214         for (; i < l; i++) {
215             this.players[i].turn = i;
216         }
217         return this;
218     },
219 
220 
221     /**
222      * Start playing in turn
223      * Player whose turn is 0 will be the first
224      */
225     startPlaying: function() {
226         this.currentTurn = 0;
227 
228         this.orderPlayers();
229 
230         this.currentPlayer = this.getPlayerByTurn(0);
231         util.log("This is turn of player 0: " + this.currentPlayer);
232 
233         this.notify({game: this, object: "Game", modified: "*", instance: this});
234 
235         this.startTimer();
236     },
237 
238 
239     /**
240      * Stop playing
241      *
242      */
243     stopPlaying: function() {
244         clearInterval(this.interval);
245         this.currentPlayer = null;
246     },
247 
248 
249     /**
250      * Get a player by attribute "id"
251      * @param id Id of player
252      * @return player
253      */
254     getPlayerById: function(id) {
255         var player = null;
256         for (var i=0 ; i<this.players.length ; i++) {
257             if(this.players[i].id == id) {
258                 player = this.players[i];
259                 break;
260             }
261         }
262 
263         return player;
264     },
265 
266 
267     /**
268      * Alias for getPlayerById
269      * @see getPlayerById
270      */
271     getPlayer: function(id) {
272         return this.getPlayerById(id);
273     },
274 
275 
276     /**
277      * Get a unit in game by attribute "id"
278      * @param id Id of unit
279      * @return unit
280      */
281     getUnitById: function(id) {
282         for (var i = 0; i < this.players.length; i++) {
283             var player = this.players[i];
284             for (var j = 0; j < player.units.length; j++) {
285                 var unit = player.units[j];
286                 if (unit.id == id) {
287                     return unit;
288                 }
289             }
290         }
291 
292         return null;
293     },
294 
295     /**
296      * Get a unit in game by attribute "cell"
297      * @param cell Cell represents position of unit
298      * @return unit
299      */
300     getUnitByCell: function(cell) {
301         for (var i = 0; i < this.players.length; i++) {
302             var player = this.players[i];
303             for (var j = 0; j < player.units.length; j++) {
304                 var unit = player.units[j];
305                 if (unit.cell == cell) {
306                     return unit;
307                 }
308             }
309         }
310         return null;
311     },
312 
313     /**
314      * Alias for getUnitById
315      * @see getUnitById
316      */
317     getUnit: function(id) {
318         return this.getUnitById(id);
319     },
320 
321 
322     /**
323      * Get a cell by coodinates x and y
324      * @param x Coordinate x of this cell
325      * @param y Coordinate y of this cell
326      * @return cell
327      */
328     getCell: function(x, y) {
329         return this.map.getCell(x, y);
330     },
331 
332     onUpdate: function(context) {
333         context.game = this;
334         this.notify(context);
335     },
336 
337 };
338 
339 module.exports = Game;
