 10 var sys     = require("sys"),
 11     http    = require('http'),
 12     io      = require('socket.io'),
 13 
 14     Client  = require("./client");
 15 
 16 /**
 17  * Class ComManager
 18  *
 19  * @constructor
 20  */
 21 function ComManager(gameEngine) {
 22 
 23     this.gameEngine = gameEngine;
 24 
 25     this.server = http.createServer(function(req, res){});
 26     this.socket = io.listen(this.server);
 27 
 28     this.clients = [];
 29 
 30     this.socket.on('connection', function(conn){
 31         this.onConnect(conn);
 32     }.bind(this));
 33 
 34 }
 35 
 36 ComManager.prototype = {
 37 
 38     /**
 39      * Server listens on a port
 40      * @param port Port Number
 41      * @return this
 42      */
 43     listen : function(port){
 44         this.server.listen(port);
 45         sys.log("Server created. Listening on port " + port);
 46         return this;
 47     },
 48 
 49 
 50     /**
 51      * Create a connection which is correspondant with a Client
 52      * @param connection Socket Connection of a Client
 53      * @return this
 54      */
 55     onConnect : function(connection) {
 56         sys.log("New connection: " + connection.sessionId);
 57         this.clients[connection.sessionId] = new Client(connection, this, this.gameEngine);
 58         return this;
 59     },
 60 
 61     /**
 62      * Send a message to a client or a group of clients.
 63      * @param to Can be a client ID or an array of client ID.
 64      * @param message Message to send to each specified client.
 65      * @return this
 66      */
 67     send: function(to, message) {
 68         // If to is an array, send message to all specified clients.
 69         if (to instanceof Array) {
 70             var i = 0, l = to.length;
 71             for (; i < l; i++) {
 72                 // TODO: verifying type of this.clients[i], if client is dead.
 73                 this.clients[to[i]].send(message);
 74             }
 75         }
 76         // Else send to the specified client.
 77         else {
 78             this.clients[to].send(message);
 79         }
 80         return this;
 81     },
 82 
 83 }
 84 
 85 module.exports = ComManager;
