 10 var util        = require("util"),
 11     Unit        = require("./unit.js"),
 12     Legacy      = require("../util/legacy.js"),
 13     Subject     = require("../util/subject.js");
 14 
 15 var legacy = new Legacy();
 16 
 17 /**
 18  * Class UnitFactory
 19  *
 20  * @author Youness HAMRI - youness.hamri@gmail.com / duc ....
 21  * @constructor
 22  */
 23 function UnitFactory(gameEngine) {
 24     this.gameEngine = gameEngine;
 25 
 26     this.currentId = 0;
 27     this.types = [];
 28 
 29     // For testing purpose only
 30     this.addUnitType("choucroute", 100, 10, 10, 3, 2, 1, []);
 31 }
 32 
 33 UnitFactory.prototype = {
 34 
 35     /**
 36      * Create a new unit.
 37      * @param type, the type of unit.
 38      * @param owner, the player who has creating a unit
 39      * @param cell, the cell that will be occupied by the unit
 40      * @return myUnit: the created unit.
 41      */
 42     create: function(type, owner, cell) {
 43 
 44         util.log("UnitFactory.create: cell=" + cell);
 45 
 46         var myUnit = new Unit();
 47 
 48         myUnit.id = this.currentId;
 49         myUnit.type = type;
 50         myUnit.owner = owner;
 51         myUnit.cell = cell;
 52 
 53         if (this.types[type] == null) {
 54             console.log("le type de l'unité que vous voulez créer n'existe pas");
 55         }
 56 
 57         else {
 58 
 59             myUnit.health = this.types[type].health;
 60             myUnit.attack = this.types[type].attack;
 61             myUnit.defense = this.types[type].defense;
 62             myUnit.view = this.types[type].view;
 63             myUnit.nbMaxMovements = this.types[type].movement;
 64             myUnit.movement = myUnit.nbMaxMovements;
 65             myUnit.range = this.types[type].range;
 66             myUnit.properties = this.types[type].properties;
 67 
 68         }
 69 
 70         legacy.inherits(new Subject(), myUnit);
 71 
 72         this.currentId ++;
 73         return myUnit;
 74     },
 75 
 76     addUnitType : function(type, health, attack, defense, view, movement, range, properties) {
 77 
 78         var obj = {
 79 
 80             "health"    : health,
 81             "attack"    : attack,
 82             "defense"   : defense,
 83             "view"      : view,
 84             "movement"  : movement,
 85             "range"     : range,
 86             "properties": properties
 87         };
 88 
 89         if ( this.types[type] != null ) {
 90             console.log("Erreur : le type de l'unité existe déjà");
 91         }
 92 
 93         else {
 94             this.types[type] = obj;
 95         }
 96     },
 97 }
 98 
 99 module.exports = UnitFactory;
