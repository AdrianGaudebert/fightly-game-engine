 10 var util = require("util");
 11 
 12 /**
 13  * Class Player
 14  *
 15  * @constructor
 16  */
 17 function Player() {
 18     this.id = null;
 19     this.name = null;
 20     this.turn = null;
 21     this.units = [];
 22     this.startPoint = {type : null , x :null , y:null};
 23     this.play = false;
 24 
 25     this.startPoint = null;
 26 
 27     this.unitFactory = null;
 28 };
 29 
 30 Player.prototype = {
 31 
 32     /**
 33      * Create a new unit for player : assign a movement value to the unit and notify it.
 34      * @param type. the type of unit
 35      * @return unit. the player unit
 36      */
 37     addUnit: function(type) {
 38 
 39         util.log("Player.addUnit: startPoint=" + this.startPoint);
 40 
 41         var unit = this.unitFactory.create(type, this, this.startPoint);
 42 
 43         unit.addObserver(this);
 44         this.units.push(unit);
 45 
 46         this.notify({object: "Player", modified: "units", instance: this});
 47 
 48         return unit;
 49     },
 50 
 51     /**
 52      * checks if the player has a unit
 53      * @param unitId. the unit identity
 54      * @return boolean. returns true if the player has this unit, and returns false otherwise.
 55      */
 56     hasUnit: function(unitId) {
 57         for (var i = 0; i < this.units.length; i++) {
 58             if (this.units[i].id == unitId)
 59                 return true ;
 60         }
 61         return false;
 62     },
 63 
 64     /**
 65      * Transform the player's attributs to a JSON string
 66      * @return the JSON string.
 67      */
 68     toJSON : function() {
 69         return {
 70             "id": this.id,
 71             "name": this.name,
 72             "turn": this.turn,
 73             "units": this.units,
 74         };
 75     },
 76 
 77     /**
 78      * Notify any change in the context
 79      * @param context. The player context.
 80      */
 81     onUpdate: function(context) {
 82         context.player = this;
 83         this.notify(context);
 84     },
 85 
 86     /**
 87      * Assign the maximun movement value to the player units.
 88      */
 89     resetUnits: function() {
 90         for(var i=0;i<this.units.length;i++) {
 91               this.units[i].resetMovement();
 92         }
 93     }
 94 };
 95 
 96 module.exports = Player;
