 12 var util = require('util');
 13 
 14 /**
 15  * Manage the actions
 16  * @author Maxime COLIN
 17  * @param game instance of the game
 18  * @constructor
 19  */
 20 function ActionManager(game) {
 21     this.game = game ;
 22 }
 23 
 24 ActionManager.prototype = {
 25 
 26     /**
 27      * Move a unit to a cell
 28      * @param string unitId id of the unit
 29      * @param int cellX x coord of the cell
 30      * @param int cellY y coord of the cell
 31      * @return true if the action succeed, false otherwise
 32      */
 33     moveUnit: function(playerId,unitId,cellX,cellY) {
 34 
 35         var unit = this.game.getUnit(unitId) ;
 36         util.log("ActionManager.moveUnit: unit=" +unit);
 37         var cell = this.game.getCell(cellX,cellY) ;
 38         var player = this.game.getPlayer(playerId) ;
 39 
 40         //check if the player can play
 41         if( !this.canPlay(player) ) {
 42             util.log("ActionManager.moveUnit: Error - Player cannot play now.");
 43             return false ;
 44         }
 45 
 46         //check if the player owns the unit
 47         if( !player.hasUnit(unitId) ) {
 48             util.log("ActionManager.moveUnit: Error - Player doesn't own the unit.");
 49             return false ;
 50         }
 51 
 52         var movement = this.game.map.getDistanceBetween(unit.cell, cell);
 53 
 54         //check the distance
 55         if( movement > unit.movement ) {
 56             util.log("ActionManager.moveUnit: Error - Unit has not enough movement to go to cell.");
 57             return false ;
 58         }
 59 
 60         //check if this cell is empty
 61         if( this.game.getUnitByCell(cell) != null ) {
 62             util.log("ActionManager.moveUnit: Error - This cell is not empty.");
 63             return false ;
 64         }
 65 
 66         //move the unit
 67         unit.setCell(cell) ;
 68         unit.setMovement(unit.movement - movement) ;
 69 
 70         return true ;
 71     },
 72 
 73 
 74     /**
 75      * Attack a target with a unit
 76      * @param string playerId id of the player
 77      * @param string unitId id of the unit
 78      * @param string targetId id of the target
 79      * @return true if the action succeed, false otherwise
 80      */
 81     attackUnit: function(playerId,unitId,targetId) {
 82 
 83         var unit = this.game.getUnit(unitId) ;
 84         var target = this.game.getUnit(targetId) ;
 85         var player = this.game.getPlayer(playerId) ;
 86 
 87         //check if the player can play
 88         if( !this.canPlay(player) ){
 89             util.log("ActionManager.attackUnit: Error - Player cannot play now.");
 90             return false ;
 91         }
 92 
 93         //check if the player owns the units
 94         if( !player.hasUnit(unitId) ) {
 95             util.log("ActionManager.attackUnit: Player doesn't own the unit.");
 96             return false ;
 97         }
 98 
 99         //check if the player doesn't own the target
100         if( player.hasUnit(targetId) ){
101             util.log("ActionManager.attackUnit: Player can't attack his own unit.");
102             return false ;
103         }
104 
105         //check the distance
106         if( this.game.map.getDistanceBetween(unit.cell, target.cell) > unit.range ){
107             util.log("ActionManager.attackUnit: Error - Unit has not enough movement to go to cell.");
108             return false ;
109         }
110 
111         //attack
112         target.setHealth(target.health - unit.attack) ;
113 
114         //riposte
115         if( this.game.map.getDistanceBetween(target.cell, unit.cell) <= target.range )
116             unit.setHealth(unit.health - target.defense) ;
117 
118         return true ;
119     },
120 
121 
122     /**
123      * check if the player can play
124      * @return true if he can play, false otherwise
125      */
126     canPlay: function(player) {
127 
128         //TODO
129         //return player.canPlay() ;
130 
131         return (this.game.currentPlayer.id == player.id) ;
132     },
133 
134 
135     /**
136      * Finish turn of current player (before time out)
137      * @param playerId Id of player who wants to end his turn
138      */
139     endTurn: function(playerId) {
140         var player = this.game.getPlayer(playerId) ;
141 
142         //check if the player can play
143         if( !this.canPlay(player) ) {
144             util.log("ActionManager.endTurn: Error - It's not your turn.");
145             return false ;
146         }
147 
148         this.game.changeTurn();
149     },
150 
151     /**
152      * Abandon the game
153      * @param playerId Id of player who wants to leave the game
154      */
155     abandon: function(playerId) {
156         this.game.removePlayer(playerId);
157     },
158 
159 };
160 
161 module.exports = ActionManager;
