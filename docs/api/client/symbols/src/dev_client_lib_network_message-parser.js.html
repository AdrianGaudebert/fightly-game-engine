 10 /**
 11  * Class MessageParser
 12  *
 13  * @author Adrian Gaudebert - adrian@gaudebert.fr
 14  * @constructor
 15  */
 16 MessageParser = function(ge) {
 17     this.gameEngine = ge;   // GameEngine
 18 };
 19 
 20 MessageParser.prototype = {
 21 
 22     parse: function(message) {
 23         console.log(message);
 24         var obj = JSON.parse(message);
 25         log('MessageParser.parse: '+obj);
 26 
 27         switch (obj.type) {
 28             case "query":
 29                 this.parseQuery(obj.data);
 30                 break;
 31             case "action":
 32                 this.parseAction(obj.data);
 33                 break;
 34             case "data":
 35                 this.parseData(obj.data);
 36                 break;
 37             default:
 38                 throw "Unknown message type"; // TODO: using an object of type Exception
 39         }
 40         return this;
 41     },
 42 
 43     parseQuery: function(data) {
 44         log('MessageParser.parseQuery: '+data);
 45 
 46         switch (data.response_type) {
 47             case "login":
 48                 this.gameEngine.onAuthenticationQuery();
 49                 break;
 50         }
 51         return this;
 52     },
 53 
 54     parseAction: function(data) {
 55         return this;
 56     },
 57 
 58     parseData: function(data) {
 59         switch (data.method) {
 60             case "new":
 61                 this.parseNewData(data.object, data.object_data);
 62                 break;
 63             case "update":
 64                 this.parseUpdateData(data.object, data.object_data);
 65                 break;
 66             case "delete":
 67                 this.parseDeleteData(data.object, data.object_data);
 68                 break;
 69         }
 70         this.gameEngine.invalidate();
 71         return this;
 72     },
 73 
 74     parseNewData: function(object, data) {
 75         switch (object) {
 76             case "Authentication":
 77                 this.gameEngine.onAuthenticationConfirm(data.id, data.valid);
 78                 break;
 79             case "Game":
 80                 this.gameEngine.world.gameData(data);
 81                 break;
 82             case "Player":
 83                 this.gameEngine.world.playerData(data);
 84                 break;
 85             case "Map":
 86                 this.gameEngine.world.mapData(data);
 87                 break;
 88             case "Unit":
 89                 this.gameEngine.world.unitData(data);
 90                 break;
 91         }
 92         return this;
 93     },
 94 
 95     parseUpdateData: function(object, data) {
 96         switch (object) {
 97             case "Game":
 98                 this.gameEngine.world.gameUpdate(data);
 99                 break;
100             case "Player":
101                 this.gameEngine.world.playerUpdate(data);
102                 break;
103             case "Unit":
104                 this.gameEngine.world.unitUpdate(data);
105                 break;
106         }
107         return this;
108     },
109 
110     parseDeleteData: function(object, data) {
111         return this;
112     },
113 
114 };
