 10 /**
 11  * Class GameEngine
 12  *
 13  * @author Adrian Gaudebert - adrian@gaudebert.fr
 14  */
 15 GameEngine = function() {
 16     this.comManager = null;
 17     this.messageBuilder = null;
 18     this.displayer = null;
 19     this.eventManager = null;
 20 
 21     this.world = null;
 22 
 23     this._handlers = [];
 24 };
 25 
 26 GameEngine.prototype = {
 27 
 28     /**
 29      * Initialize the GameEngine, instanciate tools.
 30      * @return this
 31      */
 32     init: function() {
 33         this.comManager = new ComManager(this);
 34 
 35         this.messageBuilder = new MessageBuilder();
 36         this.world = new WorldManager();
 37         this.eventManager = new EventManager(this);
 38 
 39         this.displayer = new Displayer(this.world, this.eventManager);
 40 
 41         return this;
 42     },
 43 
 44     /**
 45      * Start the GameEngine.
 46      * @return this
 47      */
 48     start: function() {
 49         this.comManager.init();
 50         return this;
 51     },
 52 
 53     /**
 54      * Invalidate the data, forcing to redraw the game.
 55      * @return this
 56      */
 57     invalidate: function() {
 58         this.displayer.invalidate();
 59         return this;
 60     },
 61 
 62     //---> Triggers
 63 
 64     /**
 65      * Call a event
 66      * @param event Name of event
 67      * @return this
 68      */
 69     trigger: function(event) {
 70     if(this._handlers[event]) {
 71             this._handlers[event].apply(this, arguments);
 72     }
 73     else {
 74         log("Event: " + event + " is unregistered.");
 75     }
 76     },
 77 
 78     /**
 79      * Add a event listener
 80      * @param event Name of event
 81      * @param callback Nam of function which listens this event
 82      * @return this
 83      */
 84     addListener: function(event, callback) {
 85     if(typeof(callback) == "function"){
 86             this._handlers[event] = callback;
 87     }
 88     else {
 89         log("Need a function to listen this event");
 90     }
 91 
 92         return this;
 93     },
 94 
 95     /**
 96      * Remove a event listener
 97      * @param event Name of event
 98      */
 99     removeListener: function(event) {
100         this._handlers = this._handlers.splice(event, 1);
101     },
102 
103     //---> Events
104 
105     onAuthenticationQuery: function() {
106         var login = this.displayer.getUserLogin();
107         this.comManager.send(this.messageBuilder.createAuthenticationLogin(login));
108     },
109 
110     onAuthenticationConfirm: function(id, valid) {
111         if (valid == true) {
112             this.world.player = id;
113             this.joinGame();
114         }
115         else {
116             // TODO
117         }
118     },
119 
120     joinGame: function() {
121         var gameId = this.displayer.getGameId();
122         this.comManager.send( this.messageBuilder.createJoinGameAction(gameId) );
123     },
124 
125 };
126 
127 var gameEngineInstance = new GameEngine().init().start();
128 
129 function log(msg) {
130     gameEngineInstance.displayer.log(msg);
131 }
