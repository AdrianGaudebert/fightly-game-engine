 10 /**
 11  * Class ComManager
 12  *
 13  * @author Adrian Gaudebert - adrian@gaudebert.fr
 14  * @constructor
 15  */
 16 ComManager = function(ge) {
 17     this.ge = ge;
 18 
 19     this.messageParser = null;
 20 
 21     this.host = 'localhost';
 22     this.port = 3401;
 23 
 24     this._socket = null;
 25 };
 26 
 27 ComManager.prototype = {
 28 
 29     init: function() {
 30         this.messageParser = new MessageParser(this.ge);
 31 
 32         this._socket = new io.Socket(this.host, { port: this.port });
 33         this._socket.on('connect', this._onOpen.bind(this));
 34         this._socket.on('message', this._onMessage.bind(this));
 35         this._socket.on('disconnect', this._onClose.bind(this));
 36 
 37         this._socket.connect();
 38     },
 39 
 40     _onOpen: function() {
 41         log("ComManager.onOpen");
 42     },
 43 
 44     _onMessage: function(msg) {
 45         log("ComManager.onMessage: "+msg);
 46         this.messageParser.parse(msg);
 47     },
 48 
 49     _onClose: function() {
 50         log("ComManager.onClose");
 51     },
 52 
 53     send: function(message) {
 54         this._socket.send(message);
 55     },
 56 
 57 };
