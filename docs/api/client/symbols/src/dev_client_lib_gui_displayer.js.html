 10 /**
 11  * Class Displayer
 12  *
 13  * @author Adrian Gaudebert - adrian@gaudebert.fr
 14  * @constructor
 15  */
 16 Displayer = function(world, eventManager) {
 17     this.world = world;
 18     this.eventManager = eventManager;
 19 
 20     // HTML elements
 21     this.loggerElt  = $('#log');
 22     this.gameElt    = $('#game');
 23 
 24     this.gameDisplayer = new GameDisplayer(this.world, this.eventManager);
 25 
 26     this.inGame = false;
 27 };
 28 
 29 Displayer.prototype = {
 30 
 31     log: function(msg) {
 32         // TODO
 33         // Check if the logging is activated
 34         this.loggerElt.append('<li>'+msg+'</li>');
 35     },
 36 
 37     getUserLogin: function() {
 38         // TODO
 39         // Use something better to get the credentials
 40         return prompt("Username", "player");
 41     },
 42 
 43     getGameId: function() {
 44         // TODO
 45         // Check the variable's name
 46         return window.location.search.substring(1).split("=")[1];
 47     },
 48 
 49     invalidate: function() {
 50         if (this.world.game != null) {
 51             this.displayGame();
 52         }
 53         else {
 54             this.displayState();
 55         }
 56         this.eventManager.bindAll();
 57     },
 58 
 59     reset: function() {
 60         this.gameElt.empty();
 61         return this;
 62     },
 63 
 64     displayState: function() {
 65         // TODO
 66     },
 67 
 68     displayGame: function() {
 69         var state = this.world.game.state;
 70         var players = this.world.game.players;
 71 
 72         this.reset();
 73 
 74         if (state == "waiting") {
 75             this.gameElt.append('<h1>Waiting for opponents...</h1>');
 76         }
 77         else {
 78             if (!this.inGame) {
 79                 this.gameDisplayer.init();
 80                 this.inGame = true;
 81             }
 82             this.gameDisplayer.display();
 83         }
 84 
 85         this.gameElt.append('<ul>');
 86         for (var i = 0; i < players.length; i++) {
 87             this.gameElt.append('<li>'+ players[i].name +'</li>');
 88         }
 89         this.gameElt.append('</ul>');
 90     },
 91 
 92 };
