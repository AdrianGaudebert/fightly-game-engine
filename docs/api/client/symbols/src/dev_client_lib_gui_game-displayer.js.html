 10 /**
 11  * Class GameDisplayer
 12  *
 13  * @author Adrian Gaudebert - adrian@gaudebert.fr
 14  * @constructor
 15  */
 16 function GameDisplayer(world, eventManager) {
 17     this.world = world;
 18     this.eventManager = eventManager;
 19 
 20     this.width = 800;
 21     this.height = 600;
 22     this.spriteSize = 64;
 23 
 24     this.mapIsDisplayed = false;
 25 
 26     this.mapSprite = './assets/images/map.png';
 27     this.unitSprite = './assets/images/unit.png';
 28     this.selectedSprite = './assets/images/selected.png';
 29 
 30     this.iso = Crafty.isometric.init(this.spriteSize);
 31 };
 32 
 33 GameDisplayer.prototype = {
 34     init: function() {
 35         var map = this.world.game.map;
 36 
 37         this.width = map.width * this.spriteSize + this.spriteSize / 2;
 38         this.height = ( map.height * this.spriteSize / 4 ) + ( this.spriteSize / 2 * ( ( map.height + 1 ) % 2 ) );
 39 
 40         Crafty.init(50, this.width, this.height);
 41 
 42         this.createSprites();
 43     },
 44 
 45     createSprites: function() {
 46         // Map sprite
 47         Crafty.sprite(this.spriteSize, this.mapSprite, {
 48             plains:     [0, 0, 1, 1],
 49             swamp:      [1, 0, 1, 1],
 50             mountain:   [2, 0, 1, 1],
 51         });
 52 
 53         // Unit sprite
 54         Crafty.sprite(this.spriteSize, this.unitSprite, {
 55             choucroute: [0, 0, 1, 1],
 56         });
 57 
 58         // Selected circle
 59         Crafty.sprite(this.spriteSize, this.selectedSprite, {
 60             selected: [0, 0, 1, 1],
 61         });
 62     },
 63 
 64     reset: function() {
 65         Crafty("unit").destroy();
 66         Crafty("selected").destroy();
 67         return this;
 68     },
 69 
 70     display: function() {
 71         this.reset()
 72             .displayMap()
 73             .displayUnits()
 74             .displayInterface();
 75         return this;
 76     },
 77 
 78     displayMap: function() {
 79         if (!this.mapIsDisplayed) {
 80             var map = this.world.game.map;
 81             var cellClickMap = function() { return new Crafty.polygon([32,16], [64,32], [32,48], [0,32]); };
 82 
 83             for (var y = 0; y < map.height; y++) {
 84                 for (var x = 0; x < map.width; x++) {
 85                     var cell = map.cells[x][y];
 86                     var tile = Crafty.e('2D, DOM, clickable, cell, ' + cell.type)
 87                         .cell(cell.x, cell.y, this.eventManager)
 88                         .clickable(cellClickMap(), this.eventManager.onCellClick);
 89 
 90                     this.iso.place(x, y, 0, tile);
 91                 }
 92             }
 93 
 94             this.mapIsDisplayed = true;
 95         }
 96         return this;
 97     },
 98 
 99     displayUnits: function() {
100         var players = this.world.game.players,
101             i = 0,
102             l = players.length;
103 
104         for (; i < l; i++) {
105             var pl = players[i],
106                 units = pl.units;
107 
108             var j = 0,
109                 ul = units.length;
110 
111             for (; j < ul; j++) {
112                 var unit = units[j];
113 
114                 if (this.eventManager.isUnitSelected() && unit.id == this.eventManager.selected.id) {
115                     // This unit is selected, let's display a circle under it
116                     var selected = Crafty.e('2D, DOM, selected');
117                     this.iso.place(unit.cell.x, unit.cell.y, 0, selected);
118                 }
119 
120                 var unitSprite = Crafty.e('2D, DOM, clickable, unit, ' + unit.type)
121                     .unit(unit.id, this.eventManager)
122                     .clickable(new Crafty.polygon([32,16], [64,32], [32,48], [0,32]), this.eventManager.onUnitClick);
123 
124                 this.iso.place(unit.cell.x, unit.cell.y, 0, unitSprite);
125                 console.log("Unit is placed to " + unit.cell.x + ", " + unit.cell.y);
126             }
127         }
128         return this;
129     },
130 
131     displayInterface: function() {
132         if (this.world.amIPlaying()) {
133             $("#game").append('<button id="end-turn-action">End my turn</button>');
134         }
135         $("#game").append('<p>Turn: ' + this.world.game.currentTurn + ' / ' + this.world.game.nbMaxTurns + '</p>');
136     },
137 
138 };
